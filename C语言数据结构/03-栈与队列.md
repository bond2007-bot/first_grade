## 一、栈（Stack）—— 后进先出（LIFO）

### 1. 顺序栈（静态分配）

```c
#include<stdio.h>
#include<stdlib.h>

#define MAXSIZE 100
typedef int ElemType;

typedef struct
{
    ElemType data[MAXSIZE];
    int top;
}Stack;

// 初始化
void initStack(Stack* s)
{
    s->top = -1;
}

// 判断是否为空
int isEmpty(Stack* s)
{
    if(s->top == -1)
    {
        printf("空的");
        return 1;
    }
    else return 0;
}

// 进栈/压栈
int push(Stack* s, ElemType e)
{
    if(s->top >= MAXSIZE - 1)
    {
        printf("满了");
        return 0;
    }
    s->top++;
    s->data[s->top] = e;
    return 1;
}

// 出栈
int pop(Stack* s, ElemType* e)
{
    if(s->top == -1)
    {
        printf("空的");
        return 0;
    }
    *e = s->data[s->top];
    s->top--;
    return 1;
}

// 获取栈顶元素
int getTop(Stack* s, ElemType* e)
{
    if(s->top == -1)
    {
        printf("empty");
        return 0;
    }
    *e = s->data[s->top];
    return 1;
}
```

### 3. 栈的应用：后缀表达式求值

```c
typedef enum
{
    LEFT_PARE, RIGHT_PARE,
    ADD, SUB, MUL, DIV, MOD,
    EOS, NUM
}contentType;
// 分别对应：(, ), +, -, *, /, %, \0, 数字

char expr[] = "5+6*3";

contentType getToken(char* symbol, int* index)
{
    *symbol = expr[*index];
    *index = *index + 1;
    switch(*symbol)
    {
        case '(': return LEFT_PARE;
        case ')': return RIGHT_PARE;
        case '+': return ADD;
        case '-': return SUB;
        case '*': return MUL;
        case '/': return DIV;
        case '%': return MOD;
        case '\0': return EOS;
        default: return NUM;
    }
}

// 后缀表达式求值
int eval(Stack* s)
{
    char symbol;
    int op1, op2;
    int index = 0;
    contentType token;
    token = getToken(&symbol, &index);
    ElemType result;

    while(token != EOS)
    {
        if(token == NUM)
        {
            push(s, symbol - '0');  // char转int
        }
        else
        {
            pop(s, &op2);
            pop(s, &op1);
            switch(token)
            {
                case ADD: push(s, op1 + op2); break;
                case SUB: push(s, op1 - op2); break;
                case MUL: push(s, op1 * op2); break;
                case DIV: push(s, op1 / op2); break;
                case MOD: push(s, op1 % op2); break;
                default: break;
            }
        }
        token = getToken(&symbol, &index);
    }
    pop(s, &result);
    printf("%d\n", result);
    return 1;
}
```

### 4. 中缀表达式转后缀表达式

```c
char expr1[] = "x/(i-j)*y";

void postfix(Stack* s)
{
    // 栈内优先级 和 栈外优先级（与枚举顺序对应）
    int in_stack[] = {0, 19, 12, 12, 13, 13, 13, 0};
    int out_stack[] = {20, 19, 12, 12, 13, 13, 13, 0};
    contentType token;
    int index = 0;
    push(s, EOS);
    char symbol;
    ElemType e;

    token = getToken(&symbol, &index);

    while(token != EOS)
    {
        if(token == NUM)
        {
            printf("%c", symbol);  // 操作数直接输出
        }
        else if(token == RIGHT_PARE)
        {
            while(s->data[s->top] != LEFT_PARE)
            {
                pop(s, &e);
                printf_token(e);
            }
            pop(s, &e);  // 左括号出栈但不输出
        }
        else
        {
            while(in_stack[s->data[s->top]] >= out_stack[token])
            {
                pop(s, &e);
                printf_token(e);
            }
            push(s, token);
        }
        token = getToken(&symbol, &index);
    }
}
```

> **转换规则**：
> 1. 操作数直接输出
> 2. 运算符：优先级 > 栈顶元素，入栈；≤ 栈顶，栈顶出栈并输出，然后压入新运算符
> 3. 左括号：栈外优先级最高必须入栈，栈内优先级最低
> 4. 右括号：栈顶若不为左括号则持续输出，直到左括号出栈（括号不输出）
> 5. \0 把剩余栈内元素都输出

## 二、队列（Queue）—— 先进先出（FIFO）

### 1. 顺序队列（静态分配）

```c
typedef struct
{
    ElemType data[MAXSIZE];
    int front;  // 队头，指向第一个有数据的元素
    int rear;   // 队尾，指向最后一个数据的后一个下标
}Queue;

void initQueue(Queue* Q)
{
    Q->front = 0;
    Q->rear = 0;
}

int isEmpty(Queue* Q)
{
    return Q->front == Q->rear;
}

int queueFull(Queue* Q)
{
    return (Q->rear + 1) % MAXSIZE == Q->front;
}

// 出队
ElemType dequeue(Queue* Q)
{
    if(Q->front == Q->rear)
    {
        printf("空的");
        return 0;
    }
    ElemType e = Q->data[Q->front];
    Q->front++;
    return e;
}

// 入队
int equeue(Queue* Q, ElemType e)
{
    if(!queueFull(Q)) return 0;
    Q->data[Q->rear] = e;
    Q->rear++;
    return 1;
}
```

### 2. 循环队列

```c
// 循环队列：默认有一个位置不存放数据，用于rear的指向
// front和rear改变时使用（Q->front/rear + 1）% MAXSIZE

// 循环队列-入队
int equeue(Queue* Q, ElemType e)
{
    if((Q->rear + 1) % MAXSIZE == Q->front)
    {
        printf("满了\n");
        return 0;
    }
    Q->data[Q->rear] = e;
    Q->rear = (Q->rear + 1) % MAXSIZE;
    return 1;
}
```

### 3. 队列的链式结构

```c
typedef struct QueueNode
{
    ElemType data;
    struct QueueNode* next;
}QueueNode;

typedef struct
{
    QueueNode* front;
    QueueNode* rear;  // rear指向最后一个节点
}Queue;

Queue* initQueue()
{
    Queue *q = (Queue*)malloc(sizeof(Queue));
    QueueNode *node = (QueueNode*)malloc(sizeof(QueueNode));
    node->data = 0;
    node->next = NULL;
    q->front = node;
    q->rear = node;
    return q;
}

// 入队（尾插）
void equeue(Queue* q, ElemType e)
{
    QueueNode* node = (QueueNode*)malloc(sizeof(QueueNode));
    node->data = e;
    node->next = NULL;
    q->rear->next = node;
    q->rear = node;
}

// 出队（删除头节点的下一个节点）
int dequeue(Queue* q, ElemType* e)
{
    QueueNode* node = q->front->next;
    *e = node->data;
    q->front->next = node->next;
    if(q->rear == node)
    {
        q->rear = q->front;
    }
    free(node);
    return 1;
}
```

## 三、期末考试考点

### 📌 栈的核心考点
1. **入栈/出栈序列判定**（必考大题）：
   - 给定入栈序列（如1,2,3,4），判断某个出栈序列是否合法
   - 卡特兰数：`f(n) = (2n)! / ((n+1) * n!²)`
2. **栈的应用**：
   - 括号匹配
   - 表达式求值 / 中缀转后缀（运算符优先级）
   - 递归转非递归
   - 数制转换（十进制转r进制）

### 📌 队列的核心考点
1. **循环队列**（重点）：
   - 队空：`front == rear`
   - 队满：`(rear + 1) % MAXSIZE == front`
   - 队列长度：`(rear - front + MAXSIZE) % MAXSIZE`
2. **队列的应用**：BFS、缓冲区、层次遍历

### 📌 常见题型
- **选择题**：判空/判满条件、入栈出栈序列、时间复杂度
- **填空题**：循环队列front/rear变化计算
- **简答题**：栈和队列的特性、应用场景
- **算法题**：括号匹配、表达式转换

### 📌 易错点
| 考点 | 易错点 |
|------|--------|
| 入栈/出栈序列 | 误以为只能全部入栈再全部出栈（实际可交替） |
| 循环队列队满 | 混淆 `(rear+1)%m == front` 和 `rear == front` |
| 链栈出栈 | 忘记保存原栈顶指针再释放 |
| 链队列删除 | 只有一个元素时，front和rear都要改 |
