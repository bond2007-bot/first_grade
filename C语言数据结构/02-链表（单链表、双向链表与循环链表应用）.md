## 一、单链表（Singly Linked List）

### 1. 结构体定义与初始化

```c
#include<stdio.h>
#include<stdlib.h>

typedef int ElemType;

typedef struct node
{
    ElemType data;       // 数据域
    struct node *next;   // 指针域
}Node;

// 单链表的初始化（带头节点）
Node* initList()
{
    Node* head = (Node*)malloc(sizeof(Node));
    head->data = 0;
    head->next = NULL;
    return head;
}

// 初始化节点（带数据域参数）
Node* initListWithElem(ElemType e)
{
    Node* node = (Node*)malloc(sizeof(Node));
    node->data = e;
    node->next = NULL;
    return node;
}
```

### 2. 头插法

```c
int insertHead(Node* L, ElemType e)
{
    Node* p = (Node*)malloc(sizeof(Node));
    p->data = e;
    p->next = L->next;   // 关键：先把原有头节点的下一个节点地址保存下来
    L->next = p;         // 无法通过直接方式获取的节点地址，必须先进行操作
    return 1;
}
```

### 3. 遍历链表

```c
void listNode(Node* L)
{
    Node* p = L->next;
    while(p != NULL)
    {
        printf("%d ", p->data);
        p = p->next;
    }
    printf("\n");
}
```

### 4. 尾插法

```c
// 获取尾节点
Node* get_tail(Node* L)
{
    Node* p = L;
    while(p->next != NULL)
    {
        p = p->next;
    }
    return p;
}

// 尾插法
Node* insertTail(Node* tail, ElemType e)
{
    Node* p = (Node*)malloc(sizeof(Node));
    p->data = e;
    tail->next = p;
    p->next = NULL;
    return p;
}

// 尾插法（节点）
Node* insertTailWithNode(Node* tail, Node *node)
{
    tail->next = node;
    node->next = NULL;
    return node;
}
```

### 5. 指定位置插入

```c
int insertNode(Node* L, int pos, ElemType e)
{
    Node* p = L;
    int i = 0;
    // 找到要插入位置的前一个节点
    while(i < pos - 1)
    {
        p = p->next;
        i++;
        if(p == NULL)  // 超出已有节点数
        {
            return 0;
        }
    }
    // 插入新节点
    Node* q = (Node*)malloc(sizeof(Node));
    q->data = e;
    q->next = p->next;
    p->next = q;
    return 1;
}
```

### 6. 删除节点

```c
int deleteNode(Node* L, int pos)
{
    Node* p = L;
    int i = 0;
    while(i < pos - 1)
    {
        p = p->next;
        i++;
        if(p == NULL)
        {
            return 0;
        }
    }
    if(p->next == NULL)
    {
        printf("要删除的位置错误");
        return 0;
    }
    Node* q = p->next;
    p->next = q->next;
    free(q);
    return 1;
}
```

### 7. 获取链表长度

```c
int listLength(Node* L)
{
    Node* p = L;
    int len = 0;
    while (p != NULL)
    {
        p = p->next;
        len++;
    }
    return len;  // 包含了头节点
}
```

### 8. 释放链表

```c
// 因为单向，所以只能从前往后释放
void freeList(Node* L)
{
    Node* p = L->next;
    Node* q;
    while(p != NULL)
    {
        q = p->next;
        free(p);
        p = q;
    }
    L->next = NULL;
}
```

### 9. 查找两个链表的共同后缀起始位置（快慢指针法）

```c
Node* findIntersectionNode(Node* headA, Node* headB)
{
    if(headA == NULL || headB == NULL)
    {
        return NULL;
    }
    Node* p = headA;
    int lenA = 0, lenB = 0;

    while (p != NULL) { p = p->next; lenA++; }
    p = headB;
    while(p != NULL) { p = p->next; lenB++; }

    Node* fast, *slow;
    int step;
    if(lenA > lenB)
    {
        step = lenA - lenB;
        fast = headA;
        slow = headB;
    }
    else
    {
        step = lenB - lenA;
        fast = headB;
        slow = headA;
    }

    for(int i = 0; i < step; i++)
        fast = fast->next;

    while(fast != slow)
    {
        fast = fast->next;
        slow = slow->next;
    }
    return fast;
}
```

## 二、双向链表（Doubly Linked List）

### 1. 结构体定义与初始化

```c
#include<stdio.h>
#include<stdlib.h>
typedef int ElemType;

// 双向链表：两个指针域，一个数据域
typedef struct node
{
    ElemType data;
    struct node *prev, *next;
}Node;

Node* initList()
{
    Node* head = (Node*)malloc(sizeof(Node));
    head->data = 0;
    head->prev = NULL;
    head->next = NULL;
    return head;
}
```

### 2. 头插法

```c
int insertHead(Node* L, ElemType e)
{
    Node* p = (Node*)malloc(sizeof(Node));
    p->data = e;
    p->prev = L;
    p->next = L->next;
    if(L->next != NULL)
    {
        L->next->prev = p;
    }
    L->next = p;
    return 1;
}
```

### 3. 遍历与尾插法

```c
void listNode(Node* L)
{
    Node* p = L->next;
    while(p != NULL)
    {
        printf("%d ", p->data);
        p = p->next;
    }
    printf("\n");
}

// 尾插法
Node* insertTail(Node* tail, ElemType e)
{
    Node* p = (Node*)malloc(sizeof(Node));
    p->data = e;
    p->prev = tail;
    tail->next = p;
    p->next = NULL;
    return p;
}

// 获取尾节点
Node* get_tail(Node* L)
{
    Node* p = L;
    while(p->next != NULL)
        p = p->next;
    return p;
}
```

### 4. 指定位置插入

```c
int insertNode(Node* L, int pos, ElemType e)
{
    Node* p = L;
    int i = 0;
    while(i < pos - 1)
    {
        p = p->next;
        i++;
        if(p == NULL) return 0;
    }
    Node* q = (Node*)malloc(sizeof(Node));
    q->data = e;
    q->prev = p;
    q->next = p->next;
    p->next->prev = q;
    p->next = q;
    return 1;
}
```

### 5. 删除节点

```c
int deleteNode(Node* L, int pos)
{
    Node* p = L;
    int i = 0;
    while(i < pos - 1)
    {
        p = p->next;
        i++;
        if(p == NULL || p->next == NULL)
        {
            printf("删除的节点位置不合理");
            return 0;
        }
    }
    Node* q = p->next;
    p->next = q->next;
    p->next->prev = p;
    free(q);
    return 1;
}
```

### 6. 主函数测试

```c

    Node* tail = get_tail(list);
    tail = insertTail(tail, 1);
    tail = insertTail(tail, 2);
    tail = insertTail(tail, 3);
    tail = insertTail(tail, 4);
    tail = insertTail(tail, 5);

}
```

## 三、循环链表应用（经典算法）

### 1. 查找链表中倒数第K个节点（快慢指针）

```c
int findNodeFS(Node* L, int k)
{
    Node* fast = L->next;
    Node* slow = L->next;
    for(int i = 0; i < k; i++)
    {
        fast = fast->next;
    }
    while(fast != NULL)
    {
        fast = fast->next;
        slow = slow->next;
    }
    printf("倒数第%d个节点的值为:%d\n", k, slow->data);
    return 1;
}
```

### 2. 删除绝对值相等的重复节点

```c
void removeNode(Node *L, int n)
{
    Node* p = L;
    int* q = (int*)malloc(sizeof(int)*(n+1));

    for(int i = 0; i < n+1; i++)
        *(q+i) = 0;

    while(p->next != NULL)
    {
        int index = abs(p->next->data);
        if(*(q+index) == 0)
        {
            *(q+index) = 1;
            p = p->next;
        }
        else
        {
            Node* temp = p->next;
            p->next = temp->next;
            free(temp);
        }
    }
    free(q);
}
```

### 3. 反转链表（三指针法）

```c
Node* reverseList(Node* head)
{
    Node* first = NULL;
    Node* second = head->next;
    Node* third;

    while(second != NULL)
    {
        third = second->next;
        second->next = first;
        // 移动指针
        first = second;
        second = third;
    }
    // 最后添加头节点
    Node* hd = initList();
    hd->next = first;
    return hd;
}
```

### 4. 删除中间节点（快慢指针）

```c
int delMidNode(Node* head)
{
    Node* fast = head->next;
    Node* slow = head;
    while(fast->next != NULL && fast->next != NULL)
    {
        fast = fast->next->next;
        slow = slow->next;
    }
    Node* q = slow->next;
    slow->next = q->next;
    free(q);
    return 1;
}
```

### 5. 链表重新排序（a1, an, a2, an-1, ...）

```c
void reOrderList(Node* head)
{
    // 1. 找到中间节点，分成前后两段
    Node* fast = head->next;
    Node* slow = head;
    while (fast != NULL && fast->next != NULL)
    {
        fast = fast->next->next;
        slow = slow->next;
    }

    // 2. 反转后半段
    Node* first = NULL;
    Node* second = slow->next;
    slow->next = NULL;  // 截断
    Node* third = NULL;

    while (second != NULL)
    {
        third = second->next;
        second->next = first;
        first = second;
        second = third;
    }

    // 3. 串联：a1->an->a2->an-1...
    Node* p1 = head->next;
    Node* q1 = first;
    Node* p2, *q2;
    while(p1 != NULL && q1 != NULL)
    {
        p2 = p1->next;
        q2 = q1->next;
        p1->next = q1;
        q1->next = p2;
        p1 = p2;
        q1 = q2;
    }
}
```

### 6. 找到链表环的入口

```c
Node* findBegin(Node* head)
{
    Node* fast = head;
    Node* slow = head;

    while(fast != NULL && fast->next != NULL)
    {
        fast = fast->next->next;
        slow = slow->next;
        if(fast == slow)  // 有环
        {
            // 计算环中节点数
            Node* p = fast;
            int count = 1;
            while(p->next != slow)
            {
                count++;
                p = p->next;
            }
            // 快指针先走count步
            fast = head;
            slow = head;
            for(int i = 0; i < count; i++)
                fast = fast->next;

            while(fast != slow)
            {
                fast = fast->next;
                slow = slow->next;
            }
            return slow;
        }
    }
    return NULL;
}
```

> **单向循环链表**：最后一个节点的指针域指向头节点，整个链表形成一个环。
> 终止条件为 `p != L` 或者 `p->next != L`（L为头节点）

## 四、期末考试考点

### 📌 核心考点
1. **单链表基本操作**：插入、删除（必须能手写出代码）
2. **带头节点 vs 不带头节点的区别**
3. **各链表判空条件**：
   - 带头节点单链表：`L->next == NULL`
   - 带头节点循环单链表：`L->next == L`
   - 带头节点双向循环链表：`L->next == L && L->prior == L`
4. **快慢指针的应用**：
   - 找链表中点
   - 找倒数第K个节点
   - 判断链表是否有环及找环入口
5. **链表逆置**（三指针法/头插法）
6. **链表排序（合并有序链表、拆分）**

### 📌 常见题型
- **选择题/填空题**：判空条件、指针修改顺序、时间复杂度
- **简答题**：头节点的作用、顺序表与链表对比
- **算法题**：链表逆置、删除指定值结点、合并有序链表、找共同后缀

### 📌 头节点作用（电子科技大学2008年考题）
1. 统一空表和非空表的处理
2. 第1个位置的插入删除更方便，无需特殊处理第一个数据结点

### 📌 易错点
- 插入/删除时指针修改顺序错误（丢失后继结点）
- 忘记 `free()` 释放内存
- 循环链表忘记闭环条件
- 双向链表修改指针时忘记更新 `prev` 指针
